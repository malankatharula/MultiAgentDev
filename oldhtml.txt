<script>
  async function runAgent() {
    const task = document.getElementById('task').value.trim();
    const code_context = document.getElementById('code_context').value.trim();

    if (!task) { alert('Please enter a task.'); return; }

    const btn = document.getElementById('run-btn');
    const spinner = document.getElementById('spinner');
    const logBox = document.getElementById('log-box');
    const agentTag = document.getElementById('agent-tag');
    const output = document.getElementById('output');
    const emptyState = document.getElementById('empty-state');

    btn.disabled = true;
    spinner.style.display = 'block';
    logBox.innerHTML = '<span style="color:#a78bfa">⟳ Running agents...</span>';
    agentTag.className = 'agent-tag tag-idle';
    agentTag.textContent = 'ROUTING...';
    document.querySelectorAll('.agent-card').forEach(c => c.classList.remove('active'));

    try {
      const res = await fetch('/api/run', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ task, code_context: code_context || null })
      });

      const data = await res.json();
      const type = data.task_type;

      // highlight active agent
      document.getElementById('card-' + type)?.classList.add('active');

      // update tag
      agentTag.className = `agent-tag tag-${type}`;
      agentTag.textContent = type.toUpperCase() + ' AGENT';

      // log
      logBox.innerHTML = data.messages.map(m => `<div class="log-line">✓ ${m}</div>`).join('');

      // render markdown
      emptyState.style.display = 'none';
      output.style.display = 'block';
      output.innerHTML = marked.parse(data.result);
      output.querySelectorAll('pre code').forEach(el => hljs.highlightElement(el));

    } catch (err) {
      logBox.innerHTML = `<span style="color:var(--red)">✗ Error: ${err.message}</span>`;
    } finally {
      btn.disabled = false;
      spinner.style.display = 'none';
    }
  }

  document.getElementById('task').addEventListener('keydown', e => {
    if (e.key === 'Enter' && e.ctrlKey) runAgent();
  });
</script>
