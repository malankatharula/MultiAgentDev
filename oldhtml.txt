<script src="https://cdn.jsdelivr.net/npm/marked/marked.min.js"></script>
<script src="https://cdnjs.cloudflare.com/ajax/libs/highlight.js/11.9.0/highlight.min.js"></script>
<script>
(() => {
  const $ = (s) => document.querySelector(s);
  const taskTa = $("#task-ta");
  const ctxTa = $("#ctx-ta");
  const counter = $("#counter");
  const runBtn = $("#run-btn");
  const runLabel = $("#run-label");
  const resultInner = $("#result-inner");
  const agentTag = $("#agent-tag");
  const agentTagLabel = $("#agent-tag-label");
  const execTime = $("#exec-time");
  const execTimeVal = $("#exec-time-val");
  const logEntries = $("#log-entries");
  const logHead = $("#log-head");
  const logStatus = $("#log-status");

  const AGENTS = {
    code:  { name: "Code Agent",  color: "var(--agent-code)" },
    debug: { name: "Debug Agent", color: "var(--agent-debug)" },
    doc:   { name: "Doc Agent",   color: "var(--agent-doc)" },
    test:  { name: "Test Agent",  color: "var(--agent-test)" },
  };

  let running = false;
  let selectedAgent = null;

  // ---------- THEME ----------
  const themeBtn = $("#theme-toggle");
  const sun = $("#icon-sun"), moon = $("#icon-moon");
  const hljsLight = $("#hljs-light"), hljsDark = $("#hljs-dark");
  function applyTheme(t) {
    document.documentElement.setAttribute("data-theme", t);
    const dark = t === "dark";
    sun.style.display = dark ? "none" : "";
    moon.style.display = dark ? "" : "none";
    hljsLight.disabled = dark;
    hljsDark.disabled = !dark;
  }
  applyTheme(localStorage.getItem("theme") || "light");
  themeBtn.addEventListener("click", () => {
    const next = document.documentElement.getAttribute("data-theme") === "dark" ? "light" : "dark";
    localStorage.setItem("theme", next);
    applyTheme(next);
  });

  // ---------- CHARACTER COUNTER ----------
  function updateCounter() {
    const n = taskTa.value.length;
    counter.textContent = `${n} / 500`;
    counter.classList.toggle("warn", n > 450 && n < 500);
    counter.classList.toggle("over", n >= 500);
  }
  taskTa.addEventListener("input", updateCounter);
  updateCounter();

  // ---------- CLEAR CONTEXT ----------
  $("#clear-ctx").addEventListener("click", () => {
    ctxTa.value = "";
    ctxTa.focus();
  });

  // ---------- AGENT VISUAL STATE ----------
  function setActiveAgent(key) {
    document.querySelectorAll(".agent").forEach(el => {
      el.classList.toggle("active", el.dataset.agent === key);
    });
    if (key && AGENTS[key]) {
      agentTag.classList.remove("idle");
      agentTag.style.setProperty("--agent-color", AGENTS[key].color);
      agentTagLabel.textContent = AGENTS[key].name;
    } else {
      agentTag.classList.add("idle");
      agentTag.style.removeProperty("--agent-color");
      agentTagLabel.textContent = "No active agent";
    }
  }

  // ---------- LOG ----------
  function ts() {
    const d = new Date();
    return d.toTimeString().slice(0,8);
  }
  function log(kind, msg) {
    const span = document.createElement("div");
    span.className = "log-entry";
    const sym = kind === "ok" ? '<span class="ok">✓</span>' : kind === "err" ? '<span class="err">✗</span>' : '<span class="pend">›</span>';
    span.innerHTML = `<span class="ts">${ts()}</span>${sym}<span>${msg}</span>`;
    logEntries.appendChild(span);
    logEntries.parentElement.scrollTop = logEntries.parentElement.scrollHeight;
  }
  function clearLog() { logEntries.innerHTML = ""; }

  // ---------- COPY BUTTONS ----------
  const copyIcon = '<svg viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="1.8" stroke-linecap="round" stroke-linejoin="round"><rect x="9" y="9" width="13" height="13" rx="2"/><path d="M5 15V5a2 2 0 0 1 2-2h10"/></svg>';
  const checkIcon = '<svg viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2" stroke-linecap="round" stroke-linejoin="round"><polyline points="20 6 9 17 4 12"/></svg>';

  function injectCopyButtons() {
    resultInner.querySelectorAll("pre").forEach(pre => {
      if (pre.parentElement.classList.contains("codeblock")) return;
      const codeEl = pre.querySelector("code");
      let lang = "text";
      if (codeEl) {
        const m = (codeEl.className || "").match(/language-([\w+-]+)/);
        if (m) lang = m[1];
      }
      const wrap = document.createElement("div");
      wrap.className = "codeblock";
      const head = document.createElement("div");
      head.className = "codeblock-head";
      head.innerHTML = `<span class="codeblock-lang">${lang}</span>`;
      const btn = document.createElement("button");
      btn.type = "button";
      btn.className = "copy-btn";
      btn.innerHTML = `${copyIcon}<span>Copy</span>`;
      btn.addEventListener("click", async () => {
        const text = codeEl ? codeEl.innerText : pre.innerText;
        try {
          await navigator.clipboard.writeText(text);
        } catch {
          const ta = document.createElement("textarea");
          ta.value = text; document.body.appendChild(ta); ta.select();
          document.execCommand("copy"); ta.remove();
        }
        btn.classList.add("copied");
        btn.innerHTML = `${checkIcon}<span>Copied</span>`;
        setTimeout(() => {
          btn.classList.remove("copied");
          btn.innerHTML = `${copyIcon}<span>Copy</span>`;
        }, 2000);
      });
      head.appendChild(btn);
      pre.parentNode.insertBefore(wrap, pre);
      wrap.appendChild(head);
      wrap.appendChild(pre);
    });
  }

  // ---------- RENDER ----------
  marked.setOptions({ breaks: true, gfm: true });

  function renderEmpty() {
    resultInner.innerHTML = `
      <div class="empty">
        <svg viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-linecap="round" stroke-linejoin="round"><polyline points="4 17 10 11 4 5"/><line x1="12" y1="19" x2="20" y2="19"/></svg>
        <div class="title">Submit a task to see agent output</div>
        <div class="hint"><kbd>Ctrl</kbd>+<kbd>Enter</kbd> to run</div>
      </div>`;
  }
  function renderLoading() {
    resultInner.innerHTML = `
      <div class="skeleton">
        <div class="sk-bar sk-title"></div>
        <div class="sk-bar sk-line"></div>
        <div class="sk-bar sk-line"></div>
        <div class="sk-bar sk-line s"></div>
        <div class="sk-bar sk-block"></div>
        <div class="sk-bar sk-line"></div>
        <div class="sk-bar sk-line s"></div>
      </div>`;
  }
  function renderMarkdown(md) {
    const html = marked.parse(md || "");
    resultInner.innerHTML = `<div class="md">${html}</div>`;
    resultInner.querySelectorAll("pre code").forEach(el => {
      try { hljs.highlightElement(el); } catch {}
    });
    injectCopyButtons();
  }
  function renderError(msg) {
    resultInner.innerHTML = `<div class="error-card"><div class="t">Request failed</div><div class="m">${msg}</div></div>`;
  }

  // ---------- SUBMIT ----------
  async function submit() {
    if (running) return;
    const task = taskTa.value.trim();
    if (!task) {
      taskTa.focus();
      return;
    }
    running = true;
    runBtn.disabled = true;
    runLabel.textContent = "Running…";
    const playIcon = runBtn.querySelector("svg");
    if (playIcon) playIcon.outerHTML = '<span class="spinner" id="run-spinner"></span>';

    setActiveAgent(null);
    execTime.style.visibility = "hidden";
    logHead.classList.add("active");
    logStatus.textContent = "Working";
    clearLog();
    log("pend", "Dispatching to orchestrator");
    if (selectedAgent) {
      log("pend", `User preference: ${AGENTS[selectedAgent]?.name || selectedAgent}`);
    }
    renderLoading();

    const t0 = performance.now();
    try {
      const requestBody = { 
        task, 
        code_context: ctxTa.value,
        ...(selectedAgent && { preferred_agent: selectedAgent })
      };
      const res = await fetch("/api/run", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify(requestBody)
      });
      if (!res.ok) throw new Error(`HTTP ${res.status}`);
      const data = await res.json();
      const elapsed = Math.round(performance.now() - t0);

      // Display messages from agent graph execution
      if (data.messages && Array.isArray(data.messages)) {
        data.messages.forEach(msg => {
          log("pend", msg);
        });
      }

      const agentKey = (data.task_type || "").toLowerCase();
      if (AGENTS[agentKey]) {
        setActiveAgent(agentKey);
        log("ok", `${AGENTS[agentKey].name} completed`);
      } else {
        log("ok", `Agent completed`);
      }
      execTimeVal.textContent = `${elapsed} ms`;
      execTime.style.visibility = "visible";
      renderMarkdown(data.result || "_(no result)_");
      logStatus.textContent = "Done";
    } catch (e) {
      log("err", e.message || "Request failed");
      logStatus.textContent = "Error";
      renderError(e.message || "Unknown error");
    } finally {
      running = false;
      logHead.classList.remove("active");
      runBtn.disabled = false;
      runLabel.textContent = "Run Agent";
      const sp = document.getElementById("run-spinner");
      if (sp) sp.outerHTML = '<svg viewBox="0 0 24 24" fill="currentColor"><path d="M8 5v14l11-7z"/></svg>';
    }
  }

  runBtn.addEventListener("click", submit);
  document.addEventListener("keydown", (e) => {
    if ((e.ctrlKey || e.metaKey) && e.key === "Enter") {
      e.preventDefault();
      submit();
    }
  });

  // Agent cards: capture selection and send to API
  document.querySelectorAll(".agent").forEach(el => {
    el.addEventListener("click", () => {
      if (running) return;
      selectedAgent = el.dataset.agent;
      setActiveAgent(selectedAgent);
    });
  });
})();
</script>
</body>
</html>
